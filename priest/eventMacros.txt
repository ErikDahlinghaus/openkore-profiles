!include tu_eventMacros.txt

automacro TeleMVP {
    MobNearDist /(Phreeoni|Orc Hero|Orc Lord|Gopinich|Pharaoh|Bloody Knight|Knight Of Abyss|Abysmal Knight|Mysteltainn)/ <15
    timeout 1
    call {
        do tele
    }
}

automacro hide {
    StatusInactiveHandle EFST_HIDING
    MobNearDist /.*/ < 3
    MobNearCount > 2
    PlayerNearDist /StapoRadio/ < 3
    timeout 20
    call {
        do ss TF_HIDING
    }
}

automacro unhide {
    StatusActiveHandle EFST_HIDING
    delay 2
    call {
        do ss TF_HIDING
    }
}

macro holywater {
    do ss AL_HOLYWATER
}

automacro respawn {
    PartyMsg /(respawn)/
    call {
        if ($.PartyMsgLastName eq "gururunt") {
            do tele 2
        }
    }
}
